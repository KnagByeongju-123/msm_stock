📘 테이블 생성 SQL 보기 
create table press_inventory_items (
  id bigserial primary key,
  name text not null,
  category text, spec text,
  unit text default 'EA',
  stock int default 0, min_stock int default 0,
  price int default 0,
  location text, vendor text, memo text, photo text,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);
create table press_inventory_history (
  id bigserial primary key,
  item_id bigint references press_inventory_items(id) on delete cascade,
  item_name text, type text, qty int, memo text,
  created_at timestamptz default now()
);
alter table press_inventory_items enable row level security;
alter table press_inventory_history enable row level security;
create policy "all" on press_inventory_items for all using (true) with check (true);
create policy "all" on press_inventory_history for all using (true) with check (true);
